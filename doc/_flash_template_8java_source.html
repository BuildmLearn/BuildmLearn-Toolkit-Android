314  private Bitmap getResizedBitmap(Bitmap image, int maxSize) {
315  int width = image.getWidth();
316  int height = image.getHeight();
317 
318  float bitmapRatio = (float) width / (float) height;
319  if (bitmapRatio > 0) {
320  width = maxSize;
321  height = (int) (width / bitmapRatio);
322  } else {
323  height = maxSize;
324  width = (int) (height * bitmapRatio);
325  }
326  return Bitmap.createScaledBitmap(image, width, height, true);
327  }

342  private Intent makePhotoIntent(String title, Context context) {

354  photo.delete();
355  } catch (Exception e) {
356  Log.d(TAG, e.getMessage());
357  Toast.makeText(context, "Please check SD card! Image shot is impossible!", Toast.LENGTH_SHORT).show();
358  }

361 
362  Intent chooser = Intent.createChooser(cameraIntent, title);
363  Intent[] extraIntents = {galleryIntent};
364 
365  chooser.putExtra(Intent.EXTRA_INITIAL_INTENTS, extraIntents);
